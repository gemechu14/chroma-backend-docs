def get_customers_by_tenant(
    db: Session,
    tenant_id: str,
    home_location_id: str | None = None,
    skip: int = 0,
    limit: int = 100,
) -> list[Customer]:
    query = db.query(Customer).filter(Customer.tenant_id == tenant_id)
    if home_location_id:
        query = query.filter(Customer.home_location_id == home_location_id)
    return query.offset(skip).limit(limit).all()

def search_customers(
    db: Session,
    tenant_id: str,
    query: str,
    home_location_id: str | None = None,
    limit: int = 20,
) -> list[Customer]:
    pattern = f"%{query}%"
    base_query = (
        db.query(Customer)
        .filter(
            Customer.tenant_id == tenant_id,
            (Customer.full_name.ilike(pattern) | Customer.phone.ilike(pattern)),
        )
    )
    if home_location_id:
        base_query = base_query.filter(Customer.home_location_id == home_location_id)
    return base_query.limit(limit).all()

def create_customer(db: Session, data: CustomerCreate) -> Customer:
    customer = Customer(**data.model_dump())
    db.add(customer)
    db.commit()
    db.refresh(customer)
    return customer

def update_customer(db: Session, customer_id: str, data: CustomerUpdate) -> Customer | None:
    customer = get_customer(db, customer_id)
    if not customer:
        return None
    for field, value in data.model_dump(exclude_none=True).items():
        setattr(customer, field, value)
    db.commit()
    db.refresh(customer)
    return customer

def delete_customer(db: Session, customer_id: str) -> bool:
    customer = get_customer(db, customer_id)
    if not customer:
        return False
    db.delete(customer)
    db.commit()
    return True

def count_customers_by_tenant(
    db: Session, tenant_id: str, home_location_id: str | None = None
) -> int:
    query = db.query(func.count(Customer.id)).filter(Customer.tenant_id == tenant_id)
    if home_location_id:
        query = query.filter(Customer.home_location_id == home_location_id)
