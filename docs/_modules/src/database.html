from sqlalchemy import create_engine, text

from sqlalchemy.orm import DeclarativeBase, sessionmaker

    {"check_same_thread": False} if settings.DATABASE_URL.startswith("sqlite") else {}
)

SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

def get_db():
    """FastAPI dependency that yields a database session."""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

def ensure_tenant_is_platform_column() -> None:
    """
    Additive migration for existing DBs (create_all does not alter columns).
    Safe to run on every startup.
    """
    if settings.DATABASE_URL.startswith("sqlite"):
        with engine.begin() as conn:
            try:
                conn.execute(
                    text(
                        "ALTER TABLE tenants ADD COLUMN is_platform BOOLEAN NOT NULL DEFAULT 0"
                    )
                )
            except OperationalError:
                pass
        return
    with engine.begin() as conn:
        try:
            conn.execute(
                text(
                    "ALTER TABLE tenants ADD COLUMN IF NOT EXISTS is_platform "
                    "BOOLEAN NOT NULL DEFAULT false"
                )
            )
        except ProgrammingError:
            pass

def ensure_user_auth_columns() -> None:
    """Add auth-related columns to users table if they don't exist."""
    is_sqlite = settings.DATABASE_URL.startswith("sqlite")
    with engine.begin() as conn:
        for col_name, col_type in _USER_AUTH_COLUMNS:
            try:
                if is_sqlite:
                    conn.execute(
                        text(f"ALTER TABLE users ADD COLUMN {col_name} {col_type}")

                        text(
                            f"ALTER TABLE users ADD COLUMN IF NOT EXISTS {col_name} {col_type}"

def ensure_product_color_columns() -> None:
    """Add level, color_family, hex_code columns and migrate tone_family to enum."""
    is_sqlite = settings.DATABASE_URL.startswith("sqlite")

    _new_columns = [
        ("level", "INTEGER"),
        ("hex_code", "VARCHAR(7)"),
    ]

    if is_sqlite:
        with engine.begin() as conn:
            for col, col_type in _new_columns:
                try:
                    conn.execute(text(f"ALTER TABLE products ADD COLUMN {col} {col_type}"))
                except OperationalError:
                    pass
            try:
                conn.execute(text("ALTER TABLE products ADD COLUMN color_family VARCHAR(10)"))
            except OperationalError:
                pass
        return

    with engine.begin() as conn:
        for enum_name, values in [
            ("tone_family_enum", "'N','A','G','V','R','C','M','H','B','P','T'"),
            ("color_family_enum", "'NEUTRAL','COOL','WARM','RED','SPECIAL'"),
        ]:
            try:
                conn.execute(text(
                    f"DO $$ BEGIN "
                    f"CREATE TYPE {enum_name} AS ENUM ({values}); "
                    f"EXCEPTION WHEN duplicate_object THEN NULL; END $$;"
                ))
            except (OperationalError, ProgrammingError):
                pass

        for col, col_type in _new_columns:
            try:
                conn.execute(text(
                    f"ALTER TABLE products ADD COLUMN IF NOT EXISTS {col} {col_type}"
                ))
            except (OperationalError, ProgrammingError):
                pass

        try:
            conn.execute(text(
                "ALTER TABLE products ADD COLUMN IF NOT EXISTS "
                "color_family color_family_enum"
            ))
        except (OperationalError, ProgrammingError):
            pass

        _tone_using = """
            CASE
                WHEN tone_family IS NULL THEN NULL::tone_family_enum
                WHEN trim(tone_family::text) = 'RB' THEN 'R'::tone_family_enum
                WHEN trim(tone_family::text) = 'VV' THEN 'V'::tone_family_enum
                WHEN trim(tone_family::text) IN (
                    'N','A','G','V','R','C','M','H','B','P','T'
                ) THEN trim(tone_family::text)::tone_family_enum
                ELSE NULL::tone_family_enum
            END
            """
        try:
            conn.execute(text(
                "ALTER TABLE products ALTER COLUMN tone_family TYPE tone_family_enum "
                f"USING {_tone_using}"
            ))
        except (OperationalError, ProgrammingError):
            pass

def ensure_formula_visit_schema() -> None:
    """Create formula_visits and link columns on formulas."""
    is_sqlite = settings.DATABASE_URL.startswith("sqlite")

    if is_sqlite:
        with engine.begin() as conn:
            conn.execute(
                text(
                    """
                    CREATE TABLE IF NOT EXISTS formula_visits (
                        id VARCHAR(36) PRIMARY KEY,
                        tenant_id VARCHAR(36) NOT NULL,
                        location_id VARCHAR(36) NOT NULL,
                        customer_id VARCHAR(36) NOT NULL,
                        created_by_user_id VARCHAR(36) NOT NULL,
                        session_name VARCHAR(150),
                        service_type VARCHAR(50),
                        notes TEXT,
                        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
                        FOREIGN KEY(tenant_id) REFERENCES tenants(id) ON DELETE CASCADE,
                        FOREIGN KEY(location_id) REFERENCES locations(id),
                        FOREIGN KEY(customer_id) REFERENCES customers(id),
                        FOREIGN KEY(created_by_user_id) REFERENCES users(id)
                    )
                    """
                )
            )
            for ddl in (
                "ALTER TABLE formulas ADD COLUMN visit_id VARCHAR(36)",
                "ALTER TABLE formulas ADD COLUMN bowl_label VARCHAR(100)",
                "ALTER TABLE formulas ADD COLUMN bowl_sort_order INTEGER",
            ):
                try:
                    conn.execute(text(ddl))
                except OperationalError:
                    pass
        return

    with engine.begin() as conn:
        conn.execute(
            text(
                """
                CREATE TABLE IF NOT EXISTS formula_visits (
                    id VARCHAR(36) PRIMARY KEY,
                    tenant_id VARCHAR(36) NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
                    location_id VARCHAR(36) NOT NULL REFERENCES locations(id),
                    customer_id VARCHAR(36) NOT NULL REFERENCES customers(id),
                    created_by_user_id VARCHAR(36) NOT NULL REFERENCES users(id),
                    session_name VARCHAR(150),
                    service_type VARCHAR(50),
                    notes TEXT,
                    created_at TIMESTAMPTZ DEFAULT now() NOT NULL
                )
                """
            )
        )
        conn.execute(
            text(
                "CREATE INDEX IF NOT EXISTS ix_formula_visits_tenant_id "
                "ON formula_visits (tenant_id)"
            )
        )
        conn.execute(
            text(
                "CREATE INDEX IF NOT EXISTS ix_formula_visits_customer_id "
                "ON formula_visits (customer_id)"
            )
        )
        for ddl in (
            "ALTER TABLE formulas ADD COLUMN IF NOT EXISTS visit_id VARCHAR(36)",
            "ALTER TABLE formulas ADD COLUMN IF NOT EXISTS bowl_label VARCHAR(100)",
            "ALTER TABLE formulas ADD COLUMN IF NOT EXISTS bowl_sort_order INTEGER",
        ):
            try:
                conn.execute(text(ddl))
            except (OperationalError, ProgrammingError):
                pass
        try:
            conn.execute(
                text(
                    """
                    DO $$
                    BEGIN
                        IF NOT EXISTS (
                            SELECT 1 FROM pg_constraint WHERE conname = 'fk_formulas_visit_id'
                        ) THEN
                            ALTER TABLE formulas
                            ADD CONSTRAINT fk_formulas_visit_id
                            FOREIGN KEY (visit_id) REFERENCES formula_visits(id)
                            ON DELETE CASCADE;
                        END IF;
                    END $$;
                    """
                )
            )
        except (OperationalError, ProgrammingError):
            pass
        try:
            conn.execute(
                text("CREATE INDEX IF NOT EXISTS ix_formulas_visit_id ON formulas (visit_id)")
            )
        except (OperationalError, ProgrammingError):
            pass

def ensure_subscription_events_table() -> None:
    """Create subscription_events audit table if missing."""
    is_sqlite = settings.DATABASE_URL.startswith("sqlite")
    ddl_pg = """
        CREATE TABLE IF NOT EXISTS subscription_events (
            id VARCHAR(36) PRIMARY KEY,
            tenant_id VARCHAR(36) NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
            event_type VARCHAR(50) NOT NULL,
            source VARCHAR(30) NOT NULL DEFAULT 'system',
            stripe_event_id VARCHAR(255),
            summary VARCHAR(500) NOT NULL,
            amount_cents INTEGER,
            currency VARCHAR(10),
            billing_interval VARCHAR(20),
            subscription_status VARCHAR(30),
            details_json TEXT,
            created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
            CONSTRAINT uq_subscription_events_stripe_event_id UNIQUE (stripe_event_id)
        )
    """
    ddl_sqlite = """
        CREATE TABLE IF NOT EXISTS subscription_events (
            id VARCHAR(36) PRIMARY KEY,
            tenant_id VARCHAR(36) NOT NULL,
            event_type VARCHAR(50) NOT NULL,
            source VARCHAR(30) NOT NULL DEFAULT 'system',
            stripe_event_id VARCHAR(255) UNIQUE,
            summary VARCHAR(500) NOT NULL,
            amount_cents INTEGER,
            currency VARCHAR(10),
            billing_interval VARCHAR(20),
            subscription_status VARCHAR(30),
            details_json TEXT,
            created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (tenant_id) REFERENCES tenants(id) ON DELETE CASCADE
        )
    """
    indexes = (
        "CREATE INDEX IF NOT EXISTS ix_subscription_events_tenant_id "
        "ON subscription_events (tenant_id)",
        "CREATE INDEX IF NOT EXISTS ix_subscription_events_event_type "
        "ON subscription_events (event_type)",
        "CREATE INDEX IF NOT EXISTS ix_subscription_events_created_at "
        "ON subscription_events (created_at)",
    )
    with engine.begin() as conn:
        try:
            conn.execute(text(ddl_pg if not is_sqlite else ddl_sqlite))
        except (OperationalError, ProgrammingError):
            pass
        for idx in indexes:
            try:
                conn.execute(text(idx))
            except (OperationalError, ProgrammingError):
                pass
