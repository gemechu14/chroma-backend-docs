@router.post(
    "/bootstrap-super-admin",
    response_model=PlatformBootstrapResponse,
    status_code=status.HTTP_201_CREATED,
    summary="Bootstrap first platform SUPER_ADMIN (one-time)",
)
def bootstrap_super_admin(
    data: PlatformBootstrapRequest,
    db: Session = Depends(get_db),
):
    """
    Creates the **platform tenant** (`is_platform=true`) and the first `SUPER_ADMIN` user.

    Disabled unless `PLATFORM_BOOTSTRAP_SECRET` is set in the server environment.
    Refuses to run if any `SUPER_ADMIN` user already exists.
    """
    if not settings.PLATFORM_BOOTSTRAP_SECRET:
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail="Platform bootstrap is not configured",
        )
    if data.secret != settings.PLATFORM_BOOTSTRAP_SECRET:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Invalid bootstrap secret",
        )
    if user_crud.count_super_admin_users(db) > 0:
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail="A platform administrator already exists; bootstrap is disabled",
        )
    existing_platform = db.query(Tenant).filter(Tenant.is_platform.is_(True)).first()
    if existing_platform:
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail="A platform tenant already exists",
        )
    if user_crud.get_users_by_email(db, data.admin_email):
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail="This email is already registered",
        )

    tenant_in = TenantCreate(
        name=data.tenant_name,
        plan="platform",
        status="active",
        is_platform=True,
    )
    platform_tenant = tenant_crud.create_tenant(db, tenant_in)

    admin_in = UserCreate(
        tenant_id=platform_tenant.id,
        email=data.admin_email,
        password=data.admin_password,
        full_name=data.admin_full_name,
        role="SUPER_ADMIN",
        is_active=True,
        default_location_id=None,
    )
    admin = user_crud.create_user(db, admin_in)

    return PlatformBootstrapResponse(
        tenant_id=platform_tenant.id,
        admin_user_id=admin.id,
        message="Platform tenant and SUPER_ADMIN created. Log in with the email and password provided.",
    )

@router.get("/billing/plans", response_model=list[SubscriptionPlanAdmin])
def list_billing_plans(
    db: Session = Depends(get_db),
    _: User = Depends(require_super_admin),
):
    """All subscription plans including inactive (platform operators)."""
    return [SubscriptionPlanAdmin(**plan_crud.plan_to_admin(p)) for p in plan_crud.list_all_plans(db)]

@router.patch("/billing/plans/{interval}", response_model=SubscriptionPlanAdmin)
def update_billing_plan(
    interval: str,
    data: SubscriptionPlanUpdate,
    db: Session = Depends(get_db),
    _: User = Depends(require_super_admin),
):
    """Update a subscription plan tier."""
    plan = plan_crud.update_plan(
        db,
        interval,
        title=data.title,
        description=data.description,
        amount_cents=data.amount_cents,
        period_label=data.period_label,
        savings_label=data.savings_label,
        features=data.features,
        stripe_price_id=data.stripe_price_id,
        highlighted=data.highlighted,
        sort_order=data.sort_order,
        active=data.active,
    )
    if not plan:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Plan not found")
    return SubscriptionPlanAdmin(**plan_crud.plan_to_admin(plan))
