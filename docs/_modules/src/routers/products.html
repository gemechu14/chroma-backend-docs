@router.get("/", response_model=PaginatedResponse[ProductRead])
def list_products(
    product_line_id: str | None = None,
    tone_family: ToneFamily | None = None,
    level: int | None = Query(default=None, ge=1, le=10),
    color_family: ColorFamily | None = None,
    page: int = Query(default=1, ge=1),
    page_size: int = Query(default=20, ge=1),
    db: Session = Depends(get_db),
    _: User = Depends(get_current_user),
):
    skip = calculate_skip(page, page_size)
    filters = dict(tone_family=tone_family, level=level, color_family=color_family)
    if product_line_id:
        items = prod_crud.get_products_by_line(db, product_line_id, skip=skip, limit=page_size, **filters)
        total = prod_crud.count_products_by_line(db, product_line_id, **filters)
    else:
        items = prod_crud.get_products(db, skip=skip, limit=page_size, **filters)
        total = prod_crud.count_products(db, **filters)
    return PaginatedResponse(
        items=items,
        page=page,
        page_size=page_size,
        total=total,
        total_pages=calculate_total_pages(total, page_size),
    )

def _resolve_tenant_catalog_id(
    current_user: User,
    db: Session,
    tenant_id: str | None,
) -> str:
    if is_platform_super_admin(current_user, db):
        if not tenant_id:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="Query parameter tenant_id is required for platform administrators",
            )
        return tenant_id
    if tenant_id and tenant_id != current_user.tenant_id:
        raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Access denied")
    return current_user.tenant_id

@router.get("/tenant-catalog", response_model=PaginatedResponse[TenantProductRead])
def list_tenant_products(
    tenant_id: str | None = Query(
        default=None,
        description="Salon tenant id (required when logged in as SUPER_ADMIN).",
    ),
    enabled_only: bool = False,
    location_id: str | None = Query(
        default=None,
        description=(
            "Location id to scope catalog to products tracked at that location. "
            "Defaults to current user's default location for salon users."
        ),
    ),
    page: int = Query(default=1, ge=1),
    page_size: int = Query(default=20, ge=1),
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user),
):
    effective = _resolve_tenant_catalog_id(current_user, db, tenant_id)
    effective_location_id = location_id
    if not is_platform_super_admin(current_user, db):
        effective_location_id = location_id or current_user.default_location_id
        if not effective_location_id:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail=(
                    "location_id is required because your account has no default_location_id. "
                    "Provide ?location_id=... or set a default location."
                ),
            )
    skip = calculate_skip(page, page_size)
    if enabled_only:
        items = tp_crud.get_enabled_tenant_products(
            db, effective, location_id=effective_location_id
        )
        total = tp_crud.count_enabled_tenant_products(
            db, effective, location_id=effective_location_id
        )
    else:
        items = tp_crud.get_tenant_products(
            db, effective, location_id=effective_location_id, skip=skip, limit=page_size
        )
        total = tp_crud.count_tenant_products(
            db, effective, location_id=effective_location_id
        )
    return PaginatedResponse(
        items=items,
        page=page,
        page_size=page_size,
        total=total,
        total_pages=calculate_total_pages(total, page_size),
    )
