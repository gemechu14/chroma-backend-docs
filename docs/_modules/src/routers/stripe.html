@router.post("/checkout-session", response_model=StripeCheckoutResponse)
def create_checkout_session(
    data: StripeCheckoutRequest,
    db: Session = Depends(get_db),
    current_user: User = Depends(require_role("ADMIN")),
):
    """Start Stripe Checkout for the selected billing interval (salon admin)."""
    if not stripe_billing.stripe_enabled():
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail="Stripe billing is not configured.",
        )
    tenant = tenant_crud.get_tenant(db, current_user.tenant_id)
    if not tenant or tenant.is_platform:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Tenant not found")
    try:
        url = stripe_billing.create_checkout_session(
            db,
            tenant,
            billing_interval=data.billing_interval,
            admin_email=current_user.email,
            admin_name=current_user.full_name,
        )
    except ValueError as exc:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail=str(exc)) from exc
    except RuntimeError as exc:
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail=str(exc),
        ) from exc
    return StripeCheckoutResponse(url=url)

@router.post("/webhook")
async def stripe_webhook(request: Request, db: Session = Depends(get_db)):
    """Stripe webhook — sync subscription state to tenant rows."""
    if not settings.STRIPE_WEBHOOK_SECRET.strip():
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail="Stripe webhook secret is not configured.",
        )
    payload = await request.body()
    sig_header = request.headers.get("stripe-signature")
    if not sig_header:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Missing signature")

    try:
        event = stripe.Webhook.construct_event(
            payload,
            sig_header,
            settings.STRIPE_WEBHOOK_SECRET,
        )
    except ValueError as exc:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Invalid payload") from exc
    except stripe.SignatureVerificationError as exc:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Invalid signature") from exc

    event_type = event.get("type") if isinstance(event, dict) else getattr(event, "type", "unknown")
    stripe_event_id = event.get("id") if isinstance(event, dict) else getattr(event, "id", None)
    try:
        stripe_billing.handle_webhook_event(
            db, event, stripe_event_id=stripe_event_id
        )
    except stripe.StripeError:
        logger.exception("Stripe API error during webhook %s", event_type)
        raise HTTPException(
            status_code=status.HTTP_502_BAD_GATEWAY,
            detail="Stripe API error during webhook handling",
        ) from None
    except Exception:
        logger.exception("Stripe webhook handler failed for %s", event_type)
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail="Webhook handler failed",
        ) from None

    return {"received": True}

@router.post("/sync-subscription", response_model=TenantSubscriptionRead)
def sync_my_subscription(
    db: Session = Depends(get_db),
    current_user: User = Depends(require_role("ADMIN")),
):
    """Sync salon subscription from Stripe after checkout (fallback if webhook was delayed)."""
    if not stripe_billing.stripe_enabled():
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail="Stripe billing is not configured.",
        )
    tenant = tenant_crud.get_tenant(db, current_user.tenant_id)
    if not tenant or tenant.is_platform:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Tenant not found")
    try:
        stripe_billing.sync_tenant_from_stripe(db, tenant)
    except ValueError as exc:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail=str(exc)) from exc
    except stripe.StripeError as exc:
        raise HTTPException(
            status_code=status.HTTP_502_BAD_GATEWAY,
            detail=f"Could not sync from Stripe: {exc.user_message or str(exc)}",
        ) from exc
    db.refresh(tenant)
    return TenantSubscriptionRead(**subscription_read_from_tenant(tenant, db))
