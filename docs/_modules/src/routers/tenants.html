@router.get("/", response_model=PaginatedResponse[TenantRead])
def list_tenants(
    page: int = Query(default=1, ge=1),
    page_size: int = Query(default=20, ge=1),
    include_platform: bool = Query(
        default=False,
        description="If true, include the platform tenant in the list (super admin only).",
    ),
    db: Session = Depends(get_db),
    _: User = Depends(require_super_admin),
):
    """List salon tenants (and optionally the platform tenant). Platform administrators only."""
    skip = calculate_skip(page, page_size)
    exclude_platform = not include_platform
    items = crud.get_tenants(db, skip=skip, limit=page_size, exclude_platform=exclude_platform)
    total = crud.count_tenants(db, exclude_platform=exclude_platform)
    return PaginatedResponse(
        items=items,
        page=page,
        page_size=page_size,
        total=total,
        total_pages=calculate_total_pages(total, page_size),
    )

@router.post("/mine/subscription/cancel", response_model=TenantSubscriptionRead)
def cancel_my_subscription(
    data: SubscriptionCancelRequest,
    db: Session = Depends(get_db),
    current_user: User = Depends(require_role("ADMIN")),
):
    """Cancel auto-renewal at end of current period (salon admin)."""
    tenant = crud.get_tenant(db, current_user.tenant_id)
    if not tenant or tenant.is_platform:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Tenant not found")
    if tenant_uses_stripe_billing(tenant) and stripe_billing.stripe_enabled():
        try:
            stripe_billing.cancel_subscription_at_period_end(tenant)
        except Exception as exc:
            raise HTTPException(
                status_code=status.HTTP_502_BAD_GATEWAY,
                detail=f"Could not update Stripe subscription: {exc}",
            ) from exc
    tenant.cancel_at_period_end = data.cancel_at_period_end
    db.commit()
    db.refresh(tenant)
    return TenantSubscriptionRead(**subscription_read_from_tenant(tenant, db))

@router.post("/mine/subscription/reactivate", response_model=TenantSubscriptionRead)
def reactivate_my_subscription(
    db: Session = Depends(get_db),
    current_user: User = Depends(require_role("ADMIN")),
):
    """Turn auto-renewal back on (salon admin)."""
    tenant = crud.get_tenant(db, current_user.tenant_id)
    if not tenant or tenant.is_platform:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Tenant not found")
    if not subscription_has_access(tenant):
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="Subscription has expired. Use Renew subscription to restore access.",
        )
    if tenant_uses_stripe_billing(tenant) and stripe_billing.stripe_enabled():
        try:
            stripe_billing.reactivate_stripe_subscription(tenant)
        except Exception as exc:
            raise HTTPException(
                status_code=status.HTTP_502_BAD_GATEWAY,
                detail=f"Could not update Stripe subscription: {exc}",
            ) from exc
    tenant.cancel_at_period_end = False
    if tenant.subscription_status == "canceled":
        tenant.subscription_status = "active"
        tenant.status = "active"
    db.commit()
    db.refresh(tenant)
    return TenantSubscriptionRead(**subscription_read_from_tenant(tenant, db))

@router.post("/mine/subscription/renew", response_model=TenantSubscriptionRead)
def renew_my_subscription(
    data: SubscriptionRenewRequest,
    db: Session = Depends(get_db),
    current_user: User = Depends(require_role("ADMIN")),
):
    """Extend or restore salon subscription (salon admin). Manual renew when not on Stripe."""
    tenant = crud.get_tenant(db, current_user.tenant_id)
    if not tenant or tenant.is_platform:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Tenant not found")
    if tenant_uses_stripe_billing(tenant):
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="This salon is billed through Stripe. Use Manage billing to renew or change plan.",
        )
    try:
        renew_tenant_subscription(
            tenant,
            billing_interval=data.billing_interval,
            from_trial=tenant.subscription_status == "trialing",
        )
    except ValueError as exc:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail=str(exc)) from exc
    db.commit()
    db.refresh(tenant)
    record_subscription_event(
        db,
        tenant_id=tenant.id,
        event_type=EVENT_MANUAL_RENEW,
        source="admin",
        summary=f"Subscription manually extended ({tenant.billing_interval} plan).",
        billing_interval=tenant.billing_interval,
        subscription_status=tenant.subscription_status,
    )
    return TenantSubscriptionRead(**subscription_read_from_tenant(tenant, db))

@router.post(
    "/register",
    response_model=TenantRegistrationResponse,
    status_code=status.HTTP_201_CREATED,
    summary="Register a new tenant",
    description="Public endpoint to register a new tenant, create default location, and admin user.",
)
def register_tenant(
    data: TenantRegistration,
    db: Session = Depends(get_db),
):
    """
    Register a new tenant (salon business).

    This endpoint:
    1. Creates a new tenant
    2. Creates a default location
    3. Creates the first admin user
    4. Returns tenant, location, and admin user info

    After registration, use the admin email and password to login.
    """
    # Check if email already exists
    existing_users = user_crud.get_users_by_email(db, data.admin_email)
    if existing_users:
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail="Email already registered. Please use a different email.",
        )

    # Create tenant with dose unlimited subscription (trial, then auto-renew)
    if data.billing_interval not in VALID_BILLING_INTERVALS:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="billing_interval must be monthly, annual, or biannual",
        )
    period_start, period_end = initial_subscription_dates(db, data.billing_interval, trial=True)
    tenant_data = TenantCreate(
        name=data.tenant_name,
        plan=data.plan or "dose_unlimited",
        status="trial",
        is_platform=False,
        billing_interval=data.billing_interval,
        subscription_status="trialing",
        period_start=period_start,
        period_end=period_end,
        cancel_at_period_end=False,
    )
    tenant = crud.create_tenant(db, tenant_data)

    record_subscription_event(
        db,
        tenant_id=tenant.id,
        event_type=EVENT_TRIAL_STARTED,
        source="system",
        summary=f"{plan_crud.get_trial_days(db)}-day free trial started ({data.billing_interval} plan).",
        billing_interval=data.billing_interval,
        subscription_status="trialing",
    )

    # Create default location
    location_data = LocationCreate(
        tenant_id=tenant.id,
        name=data.location_name,
        address=data.location_address,
        timezone=data.location_timezone,
    )
    location = crud.create_location(db, location_data)

    # Create admin user
    admin_user_data = UserCreate(
        tenant_id=tenant.id,
        email=data.admin_email,
        password=data.admin_password,
        full_name=data.admin_full_name,
        role="ADMIN",
        is_active=True,
        default_location_id=location.id,
    )
    admin_user = user_crud.create_user(db, admin_user_data)

    return TenantRegistrationResponse(
        tenant=tenant,
        location=location,
        admin_user_id=admin_user.id,
        admin_email=admin_user.email,
        message="Tenant registered successfully. You can now login with your admin credentials.",
    )

@router.post("/", response_model=TenantRead, status_code=status.HTTP_201_CREATED)
def create_tenant(
    data: TenantCreate,
    db: Session = Depends(get_db),
    _: User = Depends(require_super_admin),
):
    """Create a tenant (platform administrators only). Salon tenants must use `is_platform=false`."""
    if data.is_platform:
        existing = db.query(Tenant).filter(Tenant.is_platform.is_(True)).first()
        if existing:
            raise HTTPException(
                status_code=status.HTTP_409_CONFLICT,
                detail="A platform tenant already exists",
            )
    return crud.create_tenant(db, data)
