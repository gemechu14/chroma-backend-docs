def _tenant_token_fields(user: User, db: Session) -> dict:
    tenant = db.query(Tenant).filter(Tenant.id == user.tenant_id).first()
    is_platform = bool(tenant and tenant.is_platform)
    return {
        "sub": user.id,
        "tenant_id": user.tenant_id,
        "role": user.role,
        "is_platform_tenant": is_platform,
    }


def _login_user_summary(user: User, db: Session) -> LoginUserSummary:

        full_name=user.full_name,
        role=user.role,
        tenant_id=user.tenant_id,
        is_platform_tenant=bool(tenant and tenant.is_platform),

@router.post("/login", response_model=Token)
def login(
    login_data: LoginRequest,
    db: Session = Depends(get_db),
):
    """
    Authenticate a user with email and password only.
    Searches for users by email and validates password.
    """
    users = crud.get_users_by_email(db, email=login_data.email)

    if not users:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Incorrect email or password",
            headers={"WWW-Authenticate": "Bearer"},
        )

    valid_user = None
    for user in users:
        if verify_password(login_data.password, user.password_hash) and user.is_active:
            valid_user = user
            break

    if not valid_user:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Incorrect email or password",
            headers={"WWW-Authenticate": "Bearer"},
        )

    valid_user.last_login_at = datetime.now(timezone.utc)
    db.commit()
    token_data = _tenant_token_fields(valid_user, db)
    access_token = create_access_token(token_data)
    refresh_token = create_refresh_token(token_data)
    return Token(
        access_token=access_token,
        refresh_token=refresh_token,
        user=_login_user_summary(valid_user, db),
    )

@router.post("/refresh", response_model=Token)
def refresh_token(
    data: TokenRefresh,
    db: Session = Depends(get_db),
):
    """Refresh an access token using a valid refresh token."""
    payload = decode_refresh_token(data.refresh_token)
    if not payload:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid or expired refresh token",
            headers={"WWW-Authenticate": "Bearer"},
        )
    user_id = payload.get("sub")
    tenant_id = payload.get("tenant_id")
    if not user_id or not tenant_id:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid token payload",
        )
    user = crud.get_user(db, user_id)
    if not user or not user.is_active or user.tenant_id != tenant_id:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="User not found or inactive",
        )
    token_data = _tenant_token_fields(user, db)
    access_token = create_access_token(token_data)
    new_refresh = create_refresh_token(token_data)
    return Token(
        access_token=access_token,
        refresh_token=new_refresh,
        user=_login_user_summary(user, db),
    )

@router.get("/", response_model=PaginatedResponse[UserRead])
def list_users(
    tenant_id: str | None = Query(
        default=None,
        description="Salon tenant to list users for (required for SUPER_ADMIN). Ignored for salon staff.",
    ),
    page: int = Query(default=1, ge=1),
    page_size: int = Query(default=20, ge=1),
    db: Session = Depends(get_db),
    current_user: User = Depends(require_salon_admin_or_manager_or_super),
):
    if is_platform_super_admin(current_user, db):
        if not tenant_id:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="Query parameter tenant_id is required for platform administrators",
            )
        target_tenant = tenant_id
    else:
        target_tenant = current_user.tenant_id
        if tenant_id and tenant_id != current_user.tenant_id:
            raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Access denied")

    skip = calculate_skip(page, page_size)
    items = crud.get_users_by_tenant(
        db,
        target_tenant,
        default_location_id=current_user.default_location_id,
        skip=skip,
        limit=page_size,
    )
    total = crud.count_users_by_tenant(
        db, target_tenant, default_location_id=current_user.default_location_id
    )
    return PaginatedResponse(
        items=items,
        page=page,
        page_size=page_size,
        total=total,
        total_pages=calculate_total_pages(total, page_size),
    )

@router.post("/", response_model=UserRead, status_code=status.HTTP_201_CREATED)
def create_user(
    data: UserCreate,
    db: Session = Depends(get_db),
    current_user: User = Depends(require_salon_admin_or_manager_or_super),
):
    if data.role == "SUPER_ADMIN" and not is_platform_super_admin(current_user, db):
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Only platform administrators can create SUPER_ADMIN users",
        )
    if is_platform_super_admin(current_user, db):
        if not can_access_tenant(current_user, db, data.tenant_id):
            raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Access denied")
    else:
        if current_user.tenant_id != data.tenant_id and current_user.role != "ADMIN":
            raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Access denied")
        if data.role == "SUPER_ADMIN":
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Insufficient permissions",
            )

    if data.role == "SUPER_ADMIN":
        t = db.query(Tenant).filter(Tenant.id == data.tenant_id).first()
        if not t or not t.is_platform:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="SUPER_ADMIN users must belong to the platform tenant",
            )

    existing = crud.get_user_by_email(db, email=data.email, tenant_id=data.tenant_id)
    if existing:
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail="Email already registered for this tenant",
        )

    send_invite = data.password is None
    user = crud.create_user(db, data, send_invite=send_invite)

    if send_invite and user.invite_token:
        invite_url = f"{settings.FRONTEND_URL}/set-password?token={user.invite_token}"
        send_email(
            to_email=user.email,
            subject=f"{settings.APP_NAME} — You're Invited!",
            html_body=invite_email_html(user.full_name, invite_url),
        )

    return user

@router.delete("/{user_id}", status_code=status.HTTP_204_NO_CONTENT)
def delete_user(
    user_id: str,
    db: Session = Depends(get_db),
    current_user: User = Depends(require_admin_or_super),
):
    user = crud.get_user(db, user_id)

        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Only platform administrators can remove SUPER_ADMIN users",
