def record_subscription_event(
    db: Session,
    *,
    tenant_id: str,
    event_type: str,
    summary: str,
    source: str = "system",
    stripe_event_id: str | None = None,
    amount_cents: int | None = None,
    currency: str | None = None,
    billing_interval: str | None = None,
    subscription_status: str | None = None,
    details: dict[str, Any] | None = None,
    commit: bool = True,
) -> SubscriptionEvent | None:
    """
    Append a subscription audit event. Returns None if stripe_event_id already logged (idempotent).
    """
    if stripe_event_id:
        existing = (
            db.query(SubscriptionEvent)
            .filter(SubscriptionEvent.stripe_event_id == stripe_event_id)
            .first()
        )
        if existing:
            logger.debug("Skipping duplicate stripe event %s", stripe_event_id)
            return None

    event = SubscriptionEvent(
        tenant_id=tenant_id,
        event_type=event_type,
        source=source,
        stripe_event_id=stripe_event_id,
        summary=summary[:500],
        amount_cents=amount_cents,
        currency=currency,
        billing_interval=billing_interval,
        subscription_status=subscription_status,
        details_json=json.dumps(details) if details else None,
    )
    db.add(event)
    if commit:
        db.commit()
        db.refresh(event)
    return event

def list_subscription_events(
    db: Session,
    tenant_id: str,
    *,
    skip: int = 0,
    limit: int = 50,
) -> list[SubscriptionEvent]:
    return (
        db.query(SubscriptionEvent)
        .filter(SubscriptionEvent.tenant_id == tenant_id)
        .order_by(SubscriptionEvent.created_at.desc())
        .offset(skip)
        .limit(limit)
        .all()
    )

def count_subscription_events(db: Session, tenant_id: str) -> int:
    return (
        db.query(SubscriptionEvent)
        .filter(SubscriptionEvent.tenant_id == tenant_id)
        .count()
    )
