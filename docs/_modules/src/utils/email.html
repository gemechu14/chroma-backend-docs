def _button(url: str, label: str) -> str:
    return (
        f'<table role="presentation" cellspacing="0" cellpadding="0" style="margin:28px auto;">'
        f'<tr><td style="border-radius:8px;background-color:{BRAND_COLOR};">'
        f'<a href="{url}" target="_blank" style="display:inline-block;padding:14px 36px;'
        f'color:#ffffff;text-decoration:none;font-size:15px;font-weight:600;'
        f'border-radius:8px;">{label}</a>'
        f'</td></tr></table>'
    )

class _SMTPIPv4(smtplib.SMTP):
    """SMTP client that connects over IPv4 only (avoids broken IPv6 routes on some Windows setups)."""

    def _get_socket(self, host, port, timeout):  # type: ignore[override]
        if timeout is not None and not timeout:
            raise ValueError("Non-blocking socket (timeout=0) is not supported")
        last_err: OSError | None = None
        for res in socket.getaddrinfo(host, port, socket.AF_INET, socket.SOCK_STREAM):
            af, socktype, proto, _canon, sa = res
            sock = socket.socket(af, socktype, proto)
            sock.settimeout(timeout)
            try:
                sock.connect(sa)
                return sock
            except OSError as e:
                last_err = e
                sock.close()
        if last_err:
            raise last_err
        raise OSError(f"Could not resolve {host!r} to an IPv4 address")


class _SMTPSSLIPv4(smtplib.SMTP_SSL):
    """Same as SMTP_SSL but TCP connect uses IPv4 only (see SMTP_FORCE_IPV4)."""

    def _get_socket(self, host, port, timeout):  # type: ignore[override]
        if timeout is not None and not timeout:
            raise ValueError("Non-blocking socket (timeout=0) is not supported")
        if self.debuglevel > 0:
            self._print_debug("connect:", (host, port))
        last_err: OSError | None = None
        for res in socket.getaddrinfo(host, port, socket.AF_INET, socket.SOCK_STREAM):
            af, socktype, proto, _canon, sa = res
            sock = socket.socket(af, socktype, proto)
            sock.settimeout(timeout)
            try:
                sock.connect(sa)
                return self.context.wrap_socket(sock, server_hostname=self._host)
            except OSError as e:
                last_err = e
                sock.close()
        if last_err:
            raise last_err
        raise OSError(f"Could not connect to {host!r} over IPv4")


def _smtp_password() -> str:
    """Gmail app passwords are 16 chars; strip accidental spaces from .env."""
    return "".join(settings.SMTP_PASSWORD.split())

def send_email(to_email: str, subject: str, html_body: str) -> bool:
    """Send an HTML email via SMTP. Returns True on success."""
    if not settings.SMTP_USER or not settings.SMTP_PASSWORD:
        logger.warning("SMTP credentials not configured — skipping email to %s", to_email)
        return False

    msg = MIMEMultipart("alternative")
    msg["From"] = f"{settings.MAIL_FROM_NAME} <{settings.MAIL_FROM}>"
    msg["To"] = to_email
    msg["Subject"] = subject
    msg.attach(MIMEText(html_body, "html"))

    host = settings.SMTP_SERVER
    port = settings.SMTP_PORT
    timeout = settings.SMTP_TIMEOUT
    pwd = _smtp_password()

    try:
        if settings.SMTP_USE_SSL:
            ssl_cls = _SMTPSSLIPv4 if settings.SMTP_FORCE_IPV4 else smtplib.SMTP_SSL
            with ssl_cls(host, port, timeout=timeout) as server:
                server.login(settings.SMTP_USER, pwd)
                server.sendmail(settings.MAIL_FROM, to_email, msg.as_string())
        else:
            smtp_cls = _SMTPIPv4 if settings.SMTP_FORCE_IPV4 else smtplib.SMTP
            with smtp_cls(host, port, timeout=timeout) as server:
                server.ehlo()
                server.starttls()
                server.ehlo()
                server.login(settings.SMTP_USER, pwd)
                server.sendmail(settings.MAIL_FROM, to_email, msg.as_string())
        logger.info("Email sent to %s: %s", to_email, subject)
        return True
    except TimeoutError as e:
        logger.error(
            "SMTP connection to %s:%s timed out (%s). "
            "If you are on a restricted network, try SMTP_USE_SSL=true with SMTP_PORT=465, "
            "or set SMTP_FORCE_IPV4=true, or allow outbound SMTP in your firewall/VPN.",
            host,
            port,
            e,
        )
        return False
    except OSError as e:
        logger.error(
            "SMTP connection to %s:%s failed: %s. "
            "Check firewall, VPN, and that the host/port are correct.",
            host,
            port,
            e,
        )
        return False
    except Exception:
        logger.exception("Failed to send email to %s", to_email)
        return False
