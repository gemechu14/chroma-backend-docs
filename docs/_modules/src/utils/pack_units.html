"""Pack size and tracking units for salon inventory (grams / milliliters / ounces)."""

from enum import Enum

# US customary conversions (inventory stored in g or ml only).
OZ_TO_GRAMS = 28.3495
OZ_TO_ML = 29.5735

class TrackingUnit(str, Enum):
    """Canonical unit for on-hand qty, formulas, and cost per unit."""

    G = "G"
    ML = "ML"

class PackSizeUnit(str, Enum):
    """Label on bottle / tube (may differ from tracking)."""

    G = "G"
    ML = "ML"
    OZ = "OZ"

def normalize_unit(value: str | None) -> str | None:
    if value is None:
        return None
    return value.strip().upper()

def validate_tracking_unit(value: str | None) -> str:
    normalized = normalize_unit(value) or TrackingUnit.G.value
    try:
        TrackingUnit(normalized)
    except ValueError as exc:
        raise ValueError(
            f"tracking_unit must be G or ML (got {value!r}). "
            "Use pack_size_unit for ounces."
        ) from exc
    return normalized

def validate_pack_size_unit(value: str | None) -> str | None:
    if value is None:
        return None
    normalized = normalize_unit(value)
    try:
        PackSizeUnit(normalized)
    except ValueError as exc:
        raise ValueError(
            f"pack_size_unit must be one of G, ML, OZ (got {value!r})"
        ) from exc
    return normalized

def default_tracking_for_pack(pack_unit: str | None) -> str:
    """Infer tracking unit when pack is labeled in g, ml, or oz."""
    pu = validate_pack_size_unit(pack_unit) if pack_unit else None
    if pu == PackSizeUnit.ML.value:
        return TrackingUnit.ML.value
    if pu == PackSizeUnit.G.value:
        return TrackingUnit.G.value
    if pu == PackSizeUnit.OZ.value:
        return TrackingUnit.ML.value
    return TrackingUnit.G.value

def to_canonical_qty(
    qty: float,
    unit: str | None,
    tracking_unit: str,
) -> float:
    """
    Convert a quantity in pack_size_unit to canonical g or ml for inventory storage.
    When unit matches tracking (G/ML), qty is unchanged.
    """
    if not qty or qty <= 0:
        return 0.0
    tu = validate_tracking_unit(tracking_unit)
    u = validate_pack_size_unit(unit) if unit else tu
    if u is None or u == tu:
        return float(qty)
    if u == PackSizeUnit.OZ.value:
        if tu == TrackingUnit.ML.value:
            return float(qty) * OZ_TO_ML
        return float(qty) * OZ_TO_GRAMS
    if u == PackSizeUnit.G.value and tu == TrackingUnit.ML.value:
        raise ValueError("Cannot convert pack size in grams to milliliter tracking.")
    if u == PackSizeUnit.ML.value and tu == TrackingUnit.G.value:
        raise ValueError("Cannot convert pack size in milliliters to gram tracking.")
    return float(qty)

def canonical_pack_size(
    pack_size_value: float | None,
    pack_size_unit: str | None,
    tracking_unit: str,
) -> float | None:
    """Grams or milliliters per bottle/tube from labeled pack size."""
    if pack_size_value is None or pack_size_value <= 0:
        return None
    return to_canonical_qty(
        float(pack_size_value),
        pack_size_unit or tracking_unit,
        tracking_unit,
    )

def unit_cost_per_tracking(
    bottle_cost: float,
    pack_size_value: float,
    pack_size_unit: str | None,
    tracking_unit: str,
) -> float:
    """Cost per gram or milliliter from bottle price and pack label."""
    canonical = canonical_pack_size(pack_size_value, pack_size_unit, tracking_unit)
    if not canonical or canonical <= 0:
        raise ValueError("Invalid pack size for unit cost calculation")
    return bottle_cost / canonical
