from __future__ import annotations

import logging
from calendar import monthrange
from datetime import datetime, timezone
from typing import Literal

BillingInterval = Literal["monthly", "annual", "biannual"]

MONTHLY_PRICE_USD = 22.0
ANNUAL_DISCOUNT = 0.20
BIANNUAL_DISCOUNT = 0.10
TRIAL_DAYS = 3

logger = logging.getLogger(__name__)

VALID_BILLING_INTERVALS = frozenset({"monthly", "annual", "biannual"})

def utcnow() -> datetime:
    return datetime.now(timezone.utc)




def _add_months(dt: datetime, months: int) -> datetime:
    month = dt.month - 1 + months
    year = dt.year + month // 12
    month = month % 12 + 1
    day = min(dt.day, monthrange(year, month)[1])
    return dt.replace(year=year, month=month, day=day)

def add_billing_period(start: datetime, interval: str) -> datetime:
    if interval == "monthly":
        return _add_months(start, 1)
    if interval == "biannual":
        return _add_months(start, 6)
    if interval == "annual":
        return _add_months(start, 12)
    raise ValueError(f"Unknown billing interval: {interval}")

def interval_price_usd_fallback(interval: str) -> float:
    if interval == "monthly":
        return MONTHLY_PRICE_USD
    if interval == "biannual":
        return round(MONTHLY_PRICE_USD * 6 * (1 - BIANNUAL_DISCOUNT), 2)
    if interval == "annual":
        return round(MONTHLY_PRICE_USD * 12 * (1 - ANNUAL_DISCOUNT), 2)
    raise ValueError(f"Unknown billing interval: {interval}")

def subscription_has_access(tenant) -> bool:
    """True when tenant may use the salon app."""
    if getattr(tenant, "is_platform", False):
        return True
    status = getattr(tenant, "subscription_status", None) or ""
    if status in ("active", "trialing", "past_due"):
        period_end = getattr(tenant, "period_end", None)
        if period_end is None:
            return True
        end = period_end if period_end.tzinfo else period_end.replace(tzinfo=timezone.utc)
        return utcnow() < end
    return False

def refresh_tenant_subscription(db, tenant):
    """
    Apply cancel-at-period-end and logical auto-renewal (no payment charge).
    Skipped when tenant is billed through Stripe (webhooks keep state in sync).
    Mutates and commits tenant when state changes.
    """
    if getattr(tenant, "is_platform", False):
        return tenant
    if tenant_uses_stripe_billing(tenant):
        # Heal stale local dates when Stripe checkout succeeded but webhooks lagged/failed.
        if not subscription_has_access(tenant):
            from src.services import stripe_billing

            if stripe_billing.stripe_enabled():
                try:
                    stripe_billing.sync_tenant_from_stripe(db, tenant)
                    db.refresh(tenant)
                except Exception:
                    logger.warning(
                        "Could not sync Stripe subscription for tenant %s",
                        getattr(tenant, "id", "?"),
                        exc_info=True,
                    )
        return tenant
    if not getattr(tenant, "billing_interval", None):
        return tenant

    now = utcnow()
    period_end = getattr(tenant, "period_end", None)
    if period_end is None:
        return tenant
    end = period_end if period_end.tzinfo else period_end.replace(tzinfo=timezone.utc)

    if now < end:
        return tenant

    if getattr(tenant, "cancel_at_period_end", False):
        tenant.subscription_status = "canceled"
        tenant.status = "suspended"
        db.commit()
        db.refresh(tenant)
        return tenant

    # Auto-renew: extend period
    start = end
    tenant.period_start = start
    tenant.period_end = add_billing_period(start, tenant.billing_interval)
    if tenant.subscription_status == "trialing":
        tenant.subscription_status = "active"
        tenant.status = "active"
        tenant.plan = tenant.plan or "dose_unlimited"
    elif tenant.subscription_status != "active":
        tenant.subscription_status = "active"
        tenant.status = "active"
    db.commit()
    db.refresh(tenant)
    return tenant

def initial_subscription_dates(db, interval: str, *, trial: bool = True) -> tuple[datetime, datetime]:
    start = utcnow()
    if trial:
        from datetime import timedelta
        from src.crud.subscription_plan import get_trial_days

        days = get_trial_days(db)
        end = start + timedelta(days=days)
    else:
        end = add_billing_period(start, interval)
    return start, end

def renew_tenant_subscription(
    tenant,
    *,
    billing_interval: str | None = None,
    from_trial: bool = False,
) -> None:
    """
    Manually extend subscription (admin renew — no payment charge).
    Starts from now if expired, otherwise extends from current period_end.
    """
    if getattr(tenant, "is_platform", False):
        raise ValueError("Platform tenant cannot have a subscription")

    interval = billing_interval or getattr(tenant, "billing_interval", None) or "monthly"
    if interval not in VALID_BILLING_INTERVALS:
        raise ValueError(f"Unknown billing interval: {interval}")

    now = utcnow()
    period_end = getattr(tenant, "period_end", None)
    if period_end and not from_trial:
        end = period_end if period_end.tzinfo else period_end.replace(tzinfo=timezone.utc)
        start = end if end > now else now
    else:
        start = now

    tenant.billing_interval = interval
    tenant.period_start = start
    tenant.period_end = add_billing_period(start, interval)
    tenant.subscription_status = "active"
    tenant.status = "active"
    tenant.plan = tenant.plan or "dose_unlimited"
    tenant.cancel_at_period_end = False

def subscription_read_from_tenant(tenant, db=None) -> dict:
    """Build fields for TenantSubscriptionRead."""
    from src.services.stripe_billing import stripe_enabled

    interval = getattr(tenant, "billing_interval", None)
    price_usd = None
    price_label = None
    if interval and interval in VALID_BILLING_INTERVALS:
        if db is not None:
            from src.crud.subscription_plan import interval_price_usd as db_price
            from src.crud.subscription_plan import price_label_for_interval

            price_usd = db_price(db, interval)
            price_label = price_label_for_interval(db, interval)
        if price_usd is None:
            price_usd = interval_price_usd_fallback(interval)
        if price_label is None:
            if interval == "monthly":
                price_label = f"${price_usd:.0f}/month"
            elif interval == "biannual":
                price_label = f"${price_usd:.2f} every 6 months"
            else:
                price_label = f"${price_usd:.2f}/year"

    cancel = bool(getattr(tenant, "cancel_at_period_end", False))
    uses_stripe = tenant_uses_stripe_billing(tenant)
    stripe_on = stripe_enabled()
    status = getattr(tenant, "subscription_status", None)
    is_platform = getattr(tenant, "is_platform", False)
    return {
        "plan": getattr(tenant, "plan", None),
        "billing_interval": interval,
        "subscription_status": getattr(tenant, "subscription_status", None),
        "period_start": getattr(tenant, "period_start", None),
        "period_end": getattr(tenant, "period_end", None),
        "cancel_at_period_end": cancel,
        "price_usd": price_usd,
        "price_label": price_label,
        "has_access": subscription_has_access(tenant),
        "auto_renewal": not cancel,
        "stripe_enabled": stripe_on,
        "has_stripe_billing": uses_stripe,
        "payment_setup_required": (
            stripe_on and not uses_stripe and status == "trialing" and not is_platform
        ),
    }
